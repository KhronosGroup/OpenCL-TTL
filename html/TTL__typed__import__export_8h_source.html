86static inline void __attribute__((overloadable)) TTL_import_sub_tensor(

88 TTL_event_t *event) {
89 __local void *dst_address;
90 __global void *src_address;

100 const TTL_const_ext_tensor_t import_ext_tensor = TTL_create_const_ext_tensor(
101 src_address, import_shape, const_external_tensor.layout, TTL_create_offset(), const_external_tensor.elem_size);

103 TTL_import(import_int_tensor, import_ext_tensor, event);
