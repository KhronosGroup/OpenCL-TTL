201/**
202 * @brief Internal non-API helper function to allow debugging of exports and
203 * imports
204 *
205 * @param is_export
206 * @param internal_tensor
207 * @param external_tensor
208 * @param access_type
209 * @param event
210 * @param line
211 */
212static inline void __TTL_dump_transaction(const bool is_export, const TTL_const_int_tensor_t *const internal_tensor,
213 const TTL_const_ext_tensor_t *const external_tensor, const int access_type,
214 const TTL_event_t *const event, const unsigned int line) {
215 printf(is_export ? "Export " : "Import ");
216 __TTL_dump_shape_t(&internal_tensor->shape);
217 __TTL_dump_event(event);
218 printf("AccessType: %d\n ", access_type);
219 __TTL_dump_const_ext_tensor_t(external_tensor);
220 printf("\n ");
221 __TTL_dump_const_int_tensor_t(internal_tensor);
222 printf("\n line: %d\n", line);
223}
