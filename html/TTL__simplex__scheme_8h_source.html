20/**
21 * @file
22 *
23 * TTL_simplex_buffering pipelines a pair of import and export transactions using
24 * three internal buffers, in rotation: each buffer interchangeably serves as input
25 * buffer and output buffer, such that in each iteration one buffer is used both to
26 * export then import and two buffers are used by compute for reading and writing.
27 *
28 * With simplex buffering we're only waiting for previous iterations, so DMA
29 * transactions run mostly in parallel to computation, but serially with each
30 * other. Using the same buffer both for import and export is possible allowing us
31 * to overlap exporting from and importing to the same buffer.
32 *
33 * The following table draws the pipelined actions performed in simplex buffering.
34 * It specifies which tile is processed in each iteration:
35 *
36 * | Action\\Iteration | \#-1 | \#0 | \#1 | \#2 | \#i (2:NumOfTiles-2) | \#NumOfTiles-1 | \#NumOfTiles | \#NumOfTiles+1 |
37 * |-------------------|------|-----|-----|-----|----------------------|----------------|--------------|----------------|
38 * | **WaitExport** | | | | 0 | i-2 | NumOfTiles-3 | NumOfTiles-2 | NumOfTiles-1 |
39 * | **Export** | | | 0 | 1 | i-1 | NumOfTiles-2 | NumOfTiles-1 | |
40 * | **Wait Import** | | 0 | 1 | 2 | i | NumOfTiles-1 | | |
41 * | **Import** | 0 | 1 | 2 | 3 | i+1 | | | |
42 * | **Compute** | | 0 | 1 | 2 | i | NumOfTiles-1 | | |
43 *
44 * Notice the prolog (at iteration number -1) and the 2 epilogs (at iterations
45 * number NumOfTiles and NumOfTiles+1) which add in total 3 extra iterations.
46 *
47 * @example TTL_simplex_buffering.cl
48 */

94/**
95 * @brief Create a TTL_simplex_buffering_t and begin the buffering process
96 *
97 * @param int_base1 The address of the first buffer to be used in local memory
98 * @param int_base2 The address of the second buffer to be used in local memory
99 * @param int_base3 The address of the third buffer to be used in local memory
100 * @param ext_tensor_in The external tensor to import the input data from
101 * @param ext_tensor_out The external tensor to export the output data to
102 * @param event_in A pointer to the event to use for the inward (external to
103 * internal) transfer completion
104 * @param event_out A pointer to the event to use for the inward (internal to
105 * external) transfer completion
106 * @param first_tile The first tile to fetch for the scheme
107 *
108 * Solid description of TTL_double_double_buffering_t buffering here
109 *
110 * @return The TTL_simplex_buffering_t created from the input parameters
111 *
112 * Example:
113 * @code
114 * TTL_event_t tb_e_in = TTL_get_event();
115 * TTL_event_t tb_e_out = TTL_get_event();
116 * TTL_simplex_buffering_t tb_scheme = TTL_start_simplex_buffering(
117 * ext_base_in, ext_base_out, l_buff1, l_buff2, l_buff3, ext_layout_in,
118 * ext_layout_out, &tb_e_in, &tb_e_out);
119 * @endcode
120 * \n
121 *
122 * This can be optimized and standardized using the TTL_step_buffering
123 * call.
124 *
125 * @startuml
126 *
127 * start
128 *
129 *
130 * stop
131 *
132 * @enduml
133 *
134 */
135static inline TTL_SIMPLEX_BUFFERING_TYPE __attribute__((overloadable))
136__TTL_TRACE_FN(TTL_start_simplex_buffering, TTL_local(TTL_TENSOR_TYPE *) int_base1,
137 TTL_local(TTL_TENSOR_TYPE *) int_base2, TTL_local(TTL_TENSOR_TYPE *) int_base3,
138 TTL_EXT_TENSOR_TYPE ext_tensor_in, TTL_EXT_TENSOR_TYPE ext_tensor_out, TTL_event_t *event_in,
139 TTL_event_t *event_out, TTL_tile_t first_tile) {
140 TTL_SIMPLEX_BUFFERING_TYPE result;
141
142 result.common.int_base[0] = int_base1;
143 result.common.int_base[1] = int_base2;
144 result.common.int_base[2] = int_base3;
145 result.common.ext_tensor_in = ext_tensor_in;
146 result.common.ext_tensor_out = ext_tensor_out;
147 result.event_in = event_in;
148 result.event_out = event_out;
149 result.next_exported_tile = TTL_create_empty_tile();
150
151 result.common.index = 0;
152
153 result.int_prev_imported = TTL_create_empty_int_sub_tensor(int_base1);
154
155 TTL_step_buffering(&result, first_tile, TTL_create_empty_tile() __TTL_TRACE_LINE);
156
157 return result;
158}

160static inline TTL_IO_TENSOR_TYPE __attribute__((overloadable))
161__TTL_TRACE_FN(TTL_step_buffering, TTL_SIMPLEX_BUFFERING_TYPE *simplex_buffer, TTL_tile_t tile_next_import,
162 TTL_tile_t tile_current_export) {
163 // For performance, compute everything possible before waiting for the previous operations to finish. The current
164 // index contains the tile that is to be exported, so prepare the structures before beginning the export and export.
165 const TTL_layout_t next_import_layout =
166 TTL_create_layout(tile_next_import.shape.width, tile_next_import.shape.height);
167 const TTL_INT_SUB_TENSOR_TYPE next_import_int_sub_tensor =
168 TTL_create_int_sub_tensor(simplex_buffer->common.int_base[simplex_buffer->common.index],
169 tile_next_import.shape,
170 next_import_layout,
171 *TTL_to_const_tensor(&simplex_buffer->common.ext_tensor_in),
172 tile_next_import.offset);
173 const TTL_CONST_EXT_TENSOR_TYPE next_import_ext_tensor =
174 TTL_create_const_ext_tensor(simplex_buffer->common.ext_tensor_in.base,
175 tile_next_import.shape,
176 simplex_buffer->common.ext_tensor_in.layout,
177 tile_next_import.offset,
178 simplex_buffer->common.ext_tensor_in.elem_size);
179
180 const TTL_layout_t int_export_layout = TTL_create_layout(simplex_buffer->next_exported_tile.shape.width,
181 simplex_buffer->next_exported_tile.shape.height);
182 const TTL_INT_TENSOR_TYPE int_export_tensor =
183 TTL_create_int_tensor(simplex_buffer->common.int_base[simplex_buffer->common.index],
184 simplex_buffer->next_exported_tile.shape,
185 int_export_layout,
186 simplex_buffer->common.ext_tensor_out.elem_size);
187 const TTL_EXT_TENSOR_TYPE export_to = TTL_create_ext_tensor(simplex_buffer->common.ext_tensor_out.base,
188 simplex_buffer->next_exported_tile.shape,
189 simplex_buffer->common.ext_tensor_out.layout,
190 simplex_buffer->next_exported_tile.offset,
191 simplex_buffer->common.ext_tensor_out.elem_size);
192
193 // Wait for the previous (import/export)s to complete before starting the next.
194 TTL_wait(1, simplex_buffer->event_out);
195 TTL_wait(1, simplex_buffer->event_in);
196
197 if (TTL_tile_empty(simplex_buffer->next_exported_tile) == false)
198 TTL_export(*TTL_to_const_tensor(TTL_to_void_tensor(&int_export_tensor)),
199 *TTL_to_void_tensor(&export_to),
200 simplex_buffer->event_out __TTL_TRACE_LINE);
201
202 if (TTL_tile_empty(tile_next_import) == false)
203 TTL_import_sub_tensor(*TTL_to_void_sub_tensor(&next_import_int_sub_tensor),
204 *TTL_to_void_tensor(TTL_to_const_tensor(&next_import_ext_tensor)),
205 simplex_buffer->event_in __TTL_TRACE_LINE);
206
207 // The import/export has been started for the current tile, Move to the next
208 // tile.
209 simplex_buffer->common.index =
210 (simplex_buffer->common.index + 1) % TTL_ARRAYSIZE(simplex_buffer->common.int_base); // Write to.
211
212 // Retrieve buffer imported previously to read from now.
213 const TTL_INT_SUB_TENSOR_TYPE int_curr_buff_in = simplex_buffer->int_prev_imported;
214 simplex_buffer->int_prev_imported = next_import_int_sub_tensor;
215
216 // Can write to out buffer according to size of curr_tile, rather than size
217 // recently exported.
218 const TTL_layout_t curr_int_layout =
219 TTL_create_layout(tile_current_export.shape.width, tile_current_export.shape.width);
220 const TTL_INT_SUB_TENSOR_TYPE int_curr_buff_out =
221 TTL_create_int_sub_tensor(simplex_buffer->common.int_base[simplex_buffer->common.index],
222 tile_current_export.shape,
223 curr_int_layout,
224 *TTL_to_const_tensor(&simplex_buffer->common.ext_tensor_in),
225 tile_current_export.offset);
226
227 // Save last two tiles to prevent common repeated get_tile()'s.
228 simplex_buffer->next_exported_tile = tile_current_export;
229
230 return TTL_create_io_tensors(int_curr_buff_in, int_curr_buff_out);
231}
