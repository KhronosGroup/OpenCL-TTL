33/**
34 * @brief Wait for the previous import operation to complete before beginning an
35 * import of the next tile.
36 *
37 * @param db TTL_import_double_buffering_t describing the attributes of the
38 * transfer
39 * @param next_tile A description of the tile to begin importing.
40 *
41 */
42static inline TTL_INT_SUB_TENSOR_TYPE __attribute__((overloadable))
43__TTL_TRACE_FN(TTL_step_buffering, TTL_IMPORT_DOUBLE_BUFFERING_TYPE *const db, const TTL_tile_t next_tile) {
44 // For performance, compute everything possible before waiting for the
45 // previous operations to finish.
46 const TTL_layout_t int_layout = TTL_create_layout(next_tile.shape.width, next_tile.shape.height);
47 const TTL_INT_SUB_TENSOR_TYPE import_to = TTL_create_int_sub_tensor(
48 db->common.int_base[db->common.index], next_tile.shape, int_layout, db->common.ext_tensor_in, next_tile.offset);
49 const TTL_CONST_EXT_TENSOR_TYPE import_from = TTL_create_const_ext_tensor(db->common.ext_tensor_in.base,
50 next_tile.shape,
51 db->common.ext_tensor_in.layout,
52 next_tile.offset,
53 db->common.ext_tensor_in.elem_size);
54
55 TTL_wait(1, db->event);
56
57 if (TTL_tile_empty(next_tile) == false) {
58 TTL_import_sub_tensor(import_to, import_from, db->event __TTL_TRACE_LINE);
59 }
60
61 db->common.index = (db->common.index + 1) % 2; // TTL_ARRAYSIZE(db->common.int_base);
62
63 const TTL_layout_t prev_int_layout = TTL_create_layout(db->prev_tile.shape.width, db->prev_tile.shape.height);
64 const TTL_INT_SUB_TENSOR_TYPE result = TTL_create_int_sub_tensor(db->common.int_base[db->common.index],
65 db->prev_tile.shape,
66 prev_int_layout,
67 db->common.ext_tensor_in,
68 db->prev_tile.offset);
69
70 db->prev_tile = next_tile;
71
72 return result;
73}

75/**
76 * @brief Wait for the previous import operation to complete before beginning an
77 * export of next tile.
78 *
79 * @param db A TTL_export_double_buffering_t describing the attributes of the
80 * transfer
81 * @param tile_current A description of the tile to begin exporting.
82 *
83 */
84static inline TTL_INT_SUB_TENSOR_TYPE __attribute__((overloadable))
85__TTL_TRACE_FN(TTL_step_buffering, TTL_EXPORT_DOUBLE_BUFFERING_TYPE *db, TTL_tile_t tile_current) {
86 const TTL_layout_t int_layout = TTL_create_layout(db->prev_tile.shape.width, db->prev_tile.shape.height);
87 const TTL_CONST_INT_TENSOR_TYPE export_from = TTL_create_const_int_tensor(
88 db->common.int_base[db->common.index], db->prev_tile.shape, int_layout, db->common.ext_tensor_in.elem_size);
89 const TTL_EXT_TENSOR_TYPE export_to = TTL_create_ext_tensor(db->common.ext_tensor_in.base,
90 db->prev_tile.shape,
91 db->common.ext_tensor_in.layout,
92 db->prev_tile.offset,
93 db->common.ext_tensor_in.elem_size);
94
95 TTL_wait(1, db->event);
96
97 if (TTL_tile_empty(db->prev_tile) == false)
98 TTL_export(*TTL_to_void_tensor(TTL_to_const_tensor(&export_from)),
99 *TTL_to_void_tensor(&export_to),
100 db->event __TTL_TRACE_LINE);
101
102 db->common.index = (db->common.index + 1) % 2; // TTL_ARRAYSIZE(db->common.int_base);
103 const TTL_layout_t curr_int_layout = TTL_create_layout(tile_current.shape.width, tile_current.shape.height);
104 const TTL_INT_SUB_TENSOR_TYPE result = TTL_create_int_sub_tensor(db->common.int_base[db->common.index],
105 tile_current.shape,
106 curr_int_layout,
107 *TTL_to_const_tensor(&db->common.ext_tensor_in),
108 tile_current.offset);
109 db->prev_tile = tile_current;
110
111 return result;
112}
